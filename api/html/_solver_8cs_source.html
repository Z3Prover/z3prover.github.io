151  Contract.Requires(constraints != null);

153  Contract.Requires(Contract.ForAll(ps, c => c != null));
154  Context.CheckContextMatch<BoolExpr>(constraints);
155  Context.CheckContextMatch<BoolExpr>(ps);

243  public Status Consequences(IEnumerable<BoolExpr> assumptions, IEnumerable<Expr> variables, out BoolExpr[] consequences)
244  {
245  ASTVector result = new ASTVector(Context);
246  ASTVector asms = new ASTVector(Context);
247  ASTVector vars = new ASTVector(Context);
248  foreach (var asm in assumptions) asms.Push(asm);
249  foreach (var v in variables) vars.Push(v);
250  Z3_lbool r = (Z3_lbool)Native.Z3_solver_get_consequences(Context.nCtx, NativeObject, asms.NativeObject, vars.NativeObject, result.NativeObject);
251  consequences = result.ToBoolExprArray();
252  return lboolToStatus(r);
