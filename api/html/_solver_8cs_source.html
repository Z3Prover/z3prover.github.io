192  public void AssertAndTrack(BoolExpr[] constraints, BoolExpr[] ps)
193  {
194  Contract.Requires(constraints != null);
195  Contract.Requires(Contract.ForAll(constraints, c => c != null));
196  Contract.Requires(Contract.ForAll(ps, c => c != null));
197  Context.CheckContextMatch<BoolExpr>(constraints);
198  Context.CheckContextMatch<BoolExpr>(ps);
199  if (constraints.Length != ps.Length)
200  throw new Z3Exception("Argument size mismatch");
201 
202  for (int i = 0 ; i < constraints.Length; i++)
203  Native.Z3_solver_assert_and_track(Context.nCtx, NativeObject, constraints[i].NativeObject, ps[i].NativeObject);

321  public Status Consequences(IEnumerable<BoolExpr> assumptions, IEnumerable<Expr> variables, out BoolExpr[] consequences)
322  {
323  ASTVector result = new ASTVector(Context);
324  ASTVector asms = new ASTVector(Context);
325  ASTVector vars = new ASTVector(Context);
326  foreach (var asm in assumptions) asms.Push(asm);
327  foreach (var v in variables) vars.Push(v);
328  Z3_lbool r = (Z3_lbool)Native.Z3_solver_get_consequences(Context.nCtx, NativeObject, asms.NativeObject, vars.NativeObject, result.NativeObject);
329  consequences = result.ToBoolExprArray();
330  return lboolToStatus(r);
331  }
